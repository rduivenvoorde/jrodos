
# to run all tests:

# OR just from root dir
pytest tests

cd tests

# to run 1 test from 1 test file
cd tests

# to run tests AND seeing DEBUG logging
cd tests
pytest --log-cli-level DEBUG

cd tests
pytest --log-cli-level DEBUG test_provider_base.py

# to run 1 test from 1 test file AND seeing DEBUG logging
cd tests
pytest --log-cli-level DEBUG test_provider_base.py::test_qgs_OK
